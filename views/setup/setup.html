1. Encryption keys

# generate private key
openssl genrsa -out PRIV_KEY.tmp 2048
# compute public key from private key
openssl rsa -pubout -in PRIV_KEY.tmp -out PUBL.KEY.tmp

# print keys
cat PRIV_KEY.tmp
cat PUBL.KEY.tmp

# clean up
rm PRIV_KEY.tmp
rm PUBL.KEY.tmp
